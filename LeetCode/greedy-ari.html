斐波那契数列：

  1. dp[i]：第 i 个斐波那契数值为 dp[i]
  2. 递推公式：dp[i] = dp[i-1] + dp[i-2]
  3. dp 数组初始化：dp[0] = 1, dp[1] = 1
  4. 变量顺序：从前往后 剑指 Offer 10- I. 斐波那契数列
/**
 * @param {number} n
 * @return {number}
 */
var fib = function(n) {
    let dp = [];
    dp[0] = 0;
    dp[1] = 1;
    for(let i = 2; i <= n; i++){
        dp[i] = (dp[i-1] + dp[i-2])%(10**9+7)
    }
    return dp[n];
};
